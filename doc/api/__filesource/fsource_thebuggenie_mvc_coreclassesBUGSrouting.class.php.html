  181.                             $out[$value] = true;

  187.                     {

  282.          * Generate a url based on a route

  356.         /*
  357.          * array arrayDeepMerge ( array array1 [, array array2 [, array ...]] )
  358.          *
  359.          * Like array_merge
  360.          *
  361.          *    arrayDeepMerge() merges the elements of one or more arrays together so
  362.          * that the values of one are appended to the end of the previous one. It
  363.          * returns the resulting array.
  364.          *    If the input arrays have the same string keys, then the later value for
  365.          * that key will overwrite the previous one. If, however, the arrays contain
  366.          * numeric keys, the later value will not overwrite the original value, but
  367.          * will be appended.
  368.          *    If only one array is given and the array is numerically indexed, the keys
  369.          * get reindexed in a continuous way.
  370.          *
  371.          * Different from array_merge
  372.          *    If string keys have arrays for values, these arrays will merge recursively.
  373.          */
  374.         public static function arrayDeepMerge()
  375.         {
  376.             switch (func_num_args())
  377.             {
  378.                 case 0:

  380.                 case 1:
  381.                     return func_get_arg(0);
  382.                 case 2:
  383.                     $args = func_get_args();
  384.                     $args[2] = array();
  385.                     if (is_array($args[0]) && is_array($args[1]))
  386.                     {
  387.                         foreach (array_unique(array_merge(array_keys($args[0]),array_keys($args[1]))) as $key)
  388.                         {
  389.                             $isKey0 = array_key_exists($key, $args[0]);
  390.                             $isKey1 = array_key_exists($key, $args[1]);
  391.                             if ($isKey0 && $isKey1 && is_array($args[0][$key]) && is_array($args[1][$key]))
  392.                             {
  393.                                 $args[2][$key] = self::arrayDeepMerge($args[0][$key], $args[1][$key]);
  394.                             }
  395.                             else if ($isKey0 && $isKey1)
  396.                             {
  397.                                 $args[2][$key] = $args[1][$key];
  398.                             }
  399.                             else if (!$isKey1)
  400.                             {
  401.                                 $args[2][$key] = $args[0][$key];
  402.                             }
  403.                             else if (!$isKey0)
  404.                             {
  405.                                 $args[2][$key] = $args[1][$key];
  406.                             }
  407.                         }
  408.                         return $args[2];
  409.                     }
  410.                     else
  411.                     {
  412.                         return $args[1];
  413.                     }
  414.                 default :
  415.                     $args = func_get_args();
  416.                     $args[1] = sfToolkit::arrayDeepMerge($args[0], $args[1]);
  417.                     array_shift($args);
  418.                     return call_user_func_array(array('self', 'arrayDeepMerge'), $args);
  419.                     break;
  420.             }
  421.         }
