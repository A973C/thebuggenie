  535.         public static function get($name, $module = 'core', $scope = null, $uid = 0)

  561.             }
  562.             if (!array_key_exists($name, self::$_settings[$module]))
  563.             {
  564.                 return null;
  565.                 //self::$_settings[$name] = self::_loadSetting($name, $module, BUGScontext::getScope()->getID());

  567.             if ($uid !== 0 && array_key_exists($uid, self::$_settings[$module][$name]))
  568.             {
  569.                 return self::$_settings[$module][$name][$uid];
  570.             }
  571.             else
