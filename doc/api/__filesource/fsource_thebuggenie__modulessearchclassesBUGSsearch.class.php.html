  313.         public function doSearch($search_row, $overridepermissions = false, $frommodule = false, $issaved = false, $simplesearch = false)

  325.                     }

  327.             }
  328.             else
  329.             {
  330.                 $searchtitle = "Custom search";
  331.                 $groupby = (isset($_SESSION['groupby'])) ? $_SESSION['groupby'] : 0;
  332.             }
  333.             
  334.             $this->log('building search criteria');
  335.             $searchfields = $this->getSearchFields($search_row->get(B2tSavedSearches::ID));
  336.     
  337.             $crit = new B2DBCriteria();
  338.             $crit->setDistinct();
  339.             $crit->setFromTable(B2DB::getTable('B2tIssues'));
  340.             $crit->addJoin(B2DB::getTable('B2tComments'), B2tComments::ID, B2tIssues::ID, array(array(B2tComments::TARGET_TYPE, 1)));
  341.             $crit->addJoin(B2DB::getTable('B2tIssueAffectsEdition'), B2tIssueAffectsEdition::ISSUE, B2tIssues::ID);
  342.             $crit->addJoin(B2DB::getTable('B2tIssueAffectsComponent'), B2tIssueAffectsComponent::ISSUE, B2tIssues::ID);
  343.             $crit->addJoin(B2DB::getTable('B2tIssueAffectsBuild'), B2tIssueAffectsBuild::ISSUE, B2tIssues::ID);
  344.             
  345.             $crit->addWhere(B2tIssues::DELETED, 0);
  346.             
  347.             if ($groupby != '')
  348.             {
  349.                 switch ($groupby)
  350.                 {
  351.                     case 'milestone':
  352.                         $crit->addOrderBy(B2tIssues::MILESTONE, 'desc');
  353.                         break;
  354.                     case 'product':
  355.                         $crit->addOrderBy(B2tProducts::NAME, 'asc');
  356.                         break;
  357.                     case 'issuetype':
  358.                         $crit->addOrderBy(B2tIssues::ISSUE_TYPE, 'asc');
  359.                         break;
  360.                     case 'severity':
  361.                         $crit->addOrderBy(B2tIssues::SEVERITY, 'asc');
  362.                         break;
  363.                     case 'state':
  364.                         $crit->addOrderBy(B2tIssues::STATE, 'asc');
  365.                         break;
  366.                     case 'priority':
  367.                         $crit->addOrderBy(B2tIssues::PRIORITY, 'asc');
  368.                         break;
  369.                     case 'component':
  370.                         $crit->addOrderBy(B2tIssueAffectsComponent::COMPONENT, 'asc');
  371.                         break;
  372.                     case 'edition':
  373.                         $crit->addOrderBy(B2tIssueAffectsEdition::EDITION, 'asc');
  374.                         break;
  375.                     case 'assignee':
  376.                         $crit->addOrderBy(B2tIssues::ASSIGNED_TYPE, 'asc');
  377.                         $crit->addOrderBy(B2tIssues::ASSIGNED_TO);
  378.                         break;
  379.                 }
  380.             }
  381.             if (BUGScontext::getRequest()->getParameter('sort_column') && defined(BUGScontext::getRequest()->getParameter('sort_column')))
  382.             {
  383.                 if (BUGScontext::getRequest()->getParameter('sort') == 'asc')
  384.                 {
  385.                     $crit->addOrderBy(constant(BUGScontext::getRequest()->getParameter('sort_column')), 'asc');
  386.                 }
  387.                 else
  388.                 {
  389.                     $crit->addOrderBy(constant(BUGScontext::getRequest()->getParameter('sort_column')), 'desc');
  390.                 }
  391.             }
  392.             else
  393.             {
  394.                 $crit->addOrderBy(B2tIssues::LAST_UPDATED, 'desc');
  395.             }
  396.             
  397.             foreach ($searchfields as $aSearchField)
  398.             {
  399.                 if ($aSearchField['filter_table'] != 'B2tIssues')
  400.                 {
  401.                     $crit->addJoin(B2DB::getTable($aSearchField['filter_table']), constant($aSearchField['join_issues_on']), B2tIssues::ID);
  402.                 }
  403.                 $crit->addWhere(constant($aSearchField['filter_field']), $aSearchField['value'], constant($aSearchField['operator']));
  404.                 if ($aSearchField['req_value'] != "")
  405.                 {
  406.                     $crit->addWhere(constant($aSearchField['req_value_field']), $aSearchField['req_value']);
  407.                 }
  408.             }
  409.             if ($simplesearch)
  410.             {
  411.                 $firstarg = true;
  412.                 foreach ($_SESSION['simplefilters'] as $aSearchField)
  413.                 {
  414.                     if ($firstarg == false)
  415.                     {
  416.                         $critn->addOr($aSearchField['filter_field'], $aSearchField['value'], $aSearchField['operator']);
  417.                     }
  418.                     else
  419.                     {
  420.                         $critn = $crit->returnCriterion($aSearchField['filter_field'], $aSearchField['value'], $aSearchField['operator']);
  421.                     }
  422.                     $firstarg = false;
  423.                 }
  424.                 $crit->addWhere($critn);
  425.             }
  426.             $this->log('done');
  427.             
  428.             $this->log('retrieving search results from database');
  429.             $resultset = B2DB::getTable('B2tIssues')->doSelect($crit);
  430.             $this->log('done');
  431.             
  432.             $searchresults = array();
  433.             $this->log('building resultset');
  434.             while ($row = $resultset->getNextRow())
  435.             {

  439.                 {
  440.                     $searchresults[$theissue->getID()] = $theissue;
  441.                 }
  442.             }
  443.             $this->log('done');
  444.             return array('issues' => $searchresults, 'groupby' => $groupby, 'row' => $row);
  445.         }
